checkbox-item (단일 사용)

<!-- 전체 선택 / 아이템 관련 -->
<div data-checkbox-scope> <!-- 스코프 컨테이너 -->
<%- include(`${_src}/views/components/form/checkbox-item.ejs`, {
  name: 'ckbox01',
  value: 'all',
  checkboxType: 'all',
  text="전체 선택"          // 전체 선택 체크박스
}) %>
<%- include(`${_src}/views/components/form/checkbox-item.ejs`, {
  name: 'ckbox01',
  value: 'all',
  checkboxType: 'item',
  text="항목"            // 개별 아이템 체크박스 (여러 개)
  suffix: '(필수)', // 필수 체크박스 라벨 뒤에 붙는 텍스트 (class=label-tag)
}) %>
</div>
